\begin{theorembox}[ขั้นตอนวิธีการหาร (Division Algorithm)]
	ให้ $a$ และ $b$ เป็นจำนวนเต็มโดยที่ $a \neq 0$ แล้วจะมีจำนวนเต็ม $q$ และ $r$ เพียงคู่เดียวเท่านั้น ที่ทำให้ 
	\[b=a q+r\]
	โดยที่ $0 \leq r<|a|$ จะเรียก $q$ ว่า \textbf{ผลหาร} (quotient) และ $r$ ว่า \textbf{เศษ} (remainder)   
\end{theorembox}

\begin{examplebox}
	ให้ $b=60$ และ $a=7$ จงหาจำนวนเต็ม $q$ และ $r$ ตามขั้นตอนวิธีการหาร ที่ทำให้\\
	\[b=aq+r\;\text{โดยที่ $0 \leq r<|a|$}\]
\end{examplebox}
\solution เลือก $q=8$ และ $r=4$ 

เห็นได้ว่าสอดคล้องกับ $b=aq+r$ โดยที่ $0 \leq r<|a|$ 

เนื่องจาก
\[60=7(8)+4, \quad 0 \leq 4<|7|\]

\section{การหารลงตัว}
\begin{definitionbox}
	ให้ $a$ และ $b$ เป็นจำนวนเต็มโดยที่ $a \neq 0$
	\begin{itemize}
		\item เรียก $a$ ว่าเป็น \textbf{ตัวหาร} (divisor) หรือ \textbf{ตัวประกอบ} (factor) ตัวหนึ่งของ $b$ ก็ต่อเมื่อ มีจำนวนเต็ม $q$ ที่ทำให้ $b=a q$
		\item ในกรณีที่ $a$ เป็นตัวหารของ $b$ เราจะเรียกอีกแบบหนึ่งว่า $a$ หาร $b$ ลงตัว สามารถเขียนแทนได้ด้วยสัญลักษณ์ $a \mid b$
		\item ในกรณีที่ $a$ หาร $b$ ไม่ลงตัว สามารถเขียนแทนได้ด้วยสัญลักษณ์ $a \nmid b$
	\end{itemize}
\end{definitionbox}

\begin{examplebox}
	\begin{enumerate}
		\item $13 \mid 182$ เพราะมีจำนวนเต็ม $14$ ที่ทำให้ $182=(13)(14)$
		\item $-5 \mid 30$ เพราะมีจำนวนเต็ม $-6$ ที่ทำให้ $30=(-5)(-6)$
		\item $2 \nmid 5$ เพราะไม่มีจำนวนเต็ม $c$ ที่ทำให้ $5=2c$
	\end{enumerate}    
\end{examplebox}

\subsection{สมบัติการหารลงตัวที่สำคัญ}
\begin{theorembox} \label{theorem2.2}
	ให้ $a, b$ และ $c$ เป็นจำนวนเต็ม จะได้ว่า
	\begin{enumerate}
		\item $a \mid 0$
		\item $1\mid a$
		\item $a \mid a$
		\item $a \mid 1$ ก็ต่อเมื่อ $a= \pm 1$
		\item ถ้า $a \mid b$ และ $b \mid c$ แล้ว $a \mid c$
		\item ถ้า $a \mid b$ และ $c \mid d$ แล้ว $a c \mid b d$
		\item ถ้า $a \mid(b+c)$ และ $a \mid b$ แล้ว $a \mid c$
		\item $a \mid b$ และ $b \mid a$ ก็ต่อเมื่อ $a= \pm b$
		\item ถ้า $a \mid b$ และ $b \neq 0$ แล้ว $|a| \leq|b|$
		\item ถ้า $a \mid b$ และ $a \mid c$ แล้ว $a \mid(b x+c y)$ สำหรับทุกจำนวนเต็ม $x, y$
	\end{enumerate}
\end{theorembox}

\begin{examplebox}
	\begin{enumerate}
		\item เพราะว่า $5 \mid 25$ และ $25 \mid 100$ จะได้ $5 \mid 100$
		\item เพราะว่า $-7 \mid 49$ และ $3 \mid 9$ จะได้ $-21 \mid 441$
		\item เพราะว่า $3 \mid 99$ และ $3 \mid 12$ จะได้ $3 \mid (99(-2) + 12(5))$
		\item เพราะว่า $x \mid 15$ และ $15 \mid x$ จะได้ $x = \pm 15$
	\end{enumerate}
\end{examplebox}

 จากทฤษฎีบท \ref{theorem2.2} ข้อ 10 เราสามารถขยายผลได้ดังนี้
 
\begin{corollarybox}
	สำหรับจำนวนเต็ม $a, b_1, b_2, \ldots, b_n$ ใด ๆ ที่ $a \neq 0$ จะได้ว่า 
	ถ้า $a \mid b_1, a \mid b_2, \ldots, a \mid b_n$ แล้ว 
	\[ a \mid (b_1 x_1 + b_2 x_2 + \cdots + b_n x_n) \]
	เมื่อ $x_1, x_2, \ldots, x_n$ เป็นจำนวนเต็มใด ๆ
\end{corollarybox}

\subsection{ตัวหารร่วมมาก}
\begin{definitionbox}
	ให้ $a, b$ เป็นจำนวนเต็ม ซึ่ง $a \neq 0$ หรือ $b \neq 0$ เราจะเรียกจำนวนเต็มบวก $d$ ว่า \textbf{ตัวหารร่วมมาก} (greatest common divisor) หรือ ห.ร.ม. ของ $a$ และ $b$ เขียนแทนด้วย $(a, b)$ ก็ต่อเมื่อ $d$ มีสมบัติต่อไปนี้
	\begin{enumerate}
		\item $d \mid a$ และ $d \mid b$
		\item ถ้า $c \mid a$ และ $c \mid b$ แล้ว $c \leq d$
	\end{enumerate}
\end{definitionbox}

\noindent จากบทนิยาม จะพบว่า ถ้า $a, b \in \mathbb{Z}$ ซึ่ง $a \neq 0$ หรือ $b \neq 0$ จะได้ว่า
\begin{enumerate}
	\item เราสามารถหา $(a, b)$ ได้เสมอ
	\item $(a, b)$ มีเพียงจำนวนเดียวเท่านั้น
	\item $(a, b) = (a, -b) = (-a, b) = (-a, -b) = (b, a)$
	\item $(a, 0) = |a|$ เมื่อ $a \neq 0$
	\item ถ้า $a \mid b$ แล้ว $(a, b) = |a|$
\end{enumerate}

\begin{examplebox}
	\begin{enumerate}
		\item $(24, 84) = (84, 24) = 12$
		\item $(-6, -15) = (6, 15) = 3$
		\item $(-17, 289) = (17, -289) = 17$
		\item $(540, 450) = (-450, -540) = 90$
	\end{enumerate}
\end{examplebox}

\begin{theorembox}
ตัวหารร่วมมากของจำนวนเต็ม $a$ และ $b$ จะมีเพียงตัวเดียวเท่านั้น
\end{theorembox}

\begin{examplebox}
\begin{itemize}
	\item ตัวหารที่เป็นบวกทั้งหมดของ $-12$ คือ $1, 2, 3, 4, 6$ และ $12$
	\item ตัวหารที่เป็นบวกทั้งหมดของ $30$ คือ $1, 2, 3, 5, 6, 10, 15$ และ $30$
\end{itemize}
	ดังนั้น เซตของตัวหารร่วมที่เป็นบวกคือ $\{1, 2, 3, 6\}$ จึงได้ว่า $(-12, 30) = 6$
\end{examplebox}

\begin{theorembox}
	ให้ $a, b \in \mathbb{Z}$ โดยที่ $a \neq 0$ หรือ $b \neq 0$ จะได้ว่า มี $x, y \in \mathbb{Z}$ ที่ทำให้ 
	\[ (a, b) = ax + by \]
\end{theorembox}

\begin{corollarybox}
	ให้ $a, b \in \mathbb{Z}$ ซึ่ง $a \neq 0$ หรือ $b \neq 0$ จะได้ว่าสำหรับจำนวนเต็ม $c$ ใด ๆ ถ้า $c \mid a$ และ $c \mid b$ แล้ว $c \mid (a, b)$
\end{corollarybox}

\begin{theorembox}
	ให้ $a, b \in \mathbb{Z}$ ซึ่ง $a \neq 0$ หรือ $b \neq 0$ จะได้ว่า $(a, b) = 1$ ก็ต่อเมื่อมี $x, y \in \mathbb{Z}$ ที่ทำให้ $1 = ax + by$
\end{theorembox}

\begin{theorembox}
	สำหรับจำนวนเต็ม $a_1, a_2, \ldots, a_n$ ใด ๆ $\left(a_1, a_2, \ldots, a_n\right) = 1$ ก็ต่อเมื่อมีจำนวนเต็ม $x_1, x_2, \ldots, x_n$ ที่ทำให้ 
	\[ 1 = a_1 x_1 + a_2 x_2 + \cdots + a_n x_n \]
\end{theorembox}

\begin{theorembox}
	ให้ $a, b \in \mathbb{Z}$ ซึ่ง $a \neq 0$ หรือ $b \neq 0$ และ $m$ เป็นจำนวนเต็มบวก จะได้ว่า
	\[(ma, mb) = m(a, b)\]
\end{theorembox}

\begin{theorembox}
	ถ้า $a, b \in \mathbb{Z}$ โดยที่ $d = (a, b)$ แล้ว
	\[\left(\frac{a}{d}, \frac{b}{d}\right) = 1\]
\end{theorembox}

\begin{theorembox}
	ให้ $a_1, a_2, \ldots, a_n$ เป็นจำนวนเต็ม ถ้า $d = \left(a_1, a_2, \ldots, a_n\right)$ แล้ว
	\[\left(\frac{a_1}{d}, \frac{a_2}{d}, \ldots, \frac{a_n}{d}\right) = 1\] 
\end{theorembox}

\begin{theorembox}

	\[ (a, b) = (a, b + ax) = (a + bx, b) \]
\end{theorembox}

\begin{theorembox}
	ให้ $a, b, c, m \in \mathbb{Z}$ จะได้ว่า
	\begin{enumerate}
		\item ถ้า $(a, m) = 1$ และ $(b, m) = 1$ แล้ว $(ab, m) = 1$
		\item ถ้า $(a, m) = 1$ และ $b \mid a$ แล้ว $(b, m) = 1$
		\item ถ้า $a \mid bc$ และ $(a, b) = 1$ แล้ว $a \mid c$
		\item ถ้า $a \mid c$ และ $b \mid c$ โดยที่ $(a, b) = 1$ แล้ว $ab \mid c$
	\end{enumerate}
\end{theorembox}

	\[ (a, b) = (a, r) \]
\end{theorembox}

\subsection{ตัวคูณร่วมน้อย}
\begin{definitionbox}
	ให้ $a, b \in \mathbb{Z}$ ซึ่ง $a \neq 0$ และ $b \neq 0$ เราจะเรียกจำนวนเต็มบวก $m$ ว่า \textbf{ตัวคูณร่วมน้อย} (least common multiple) หรือ ค.ร.น. ของ $a$ และ $b$ เขียนแทนด้วย $[a, b]$ ก็ต่อเมื่อ $m$ มีสมบัติต่อไปนี้
	\begin{enumerate}
		\item $a \mid m$ และ $b \mid m$
		\item สำหรับจำนวนเต็มบวก $c$ ถ้า $a \mid c$ และ $b \mid c$ แล้ว $m \leq c$
	\end{enumerate}
\end{definitionbox}

\noindent จากบทนิยาม จะพบว่า ถ้า $a, b \in \mathbb{Z}$ ซึ่ง $a \neq 0$ และ $b \neq 0$ จะได้ว่า
\begin{enumerate}
	\item เซตของตัวคูณร่วมของ $a$ และ $b$ เป็นเซตอนันต์
	\item ถ้า $m$ เป็นตัวคูณร่วมของ $a$ และ $b$ จะได้ว่า $-m$ เป็นตัวคูณร่วมของ $a$ และ $b$
	\item $[a, b] = [b, a] = [a, -b] = [-a, b] = [-a, -b]$
	\item ถ้า $a \mid b$ แล้ว $[a, b] = |b|$
\end{enumerate}

\begin{examplebox}
	\begin{enumerate}
		\item $[15, 21] = 105$
		\item $[-24, -36] = 72$
		\item $[3054, 12378] = 6,300,402$
		\item $[-12, 30] = 60$
		\item $[5, 11] = 55$
	\end{enumerate}
\end{examplebox}

\observation
\begin{enumerate}
	\item $[a, b] = [b, a]$
	\item $[a, b] = [|a|, |b|]$
	\item ถ้า $a \mid b$ แล้วจะได้ว่า $[a, b] = |b|$
\end{enumerate}

\begin{theorembox}
	ให้ $a, b \in \mathbb{Z}$ ซึ่ง $a \neq 0$ และ $b \neq 0$ จะได้ว่า สำหรับจำนวนเต็ม $c$ ใด ๆ ถ้า $a \mid c$ และ $b \mid c$ แล้ว $[a, b] \mid c$
\end{theorembox}

		\end{theorembox}

		\begin{theorembox}
			ให้ $a, b \in \mathbb{Z}$ ซึ่ง $a \neq 0$ และ $b \neq 0$ และ $m \in \mathbb{N}$ จะได้ว่า 
			\[ [ma, mb] = m[a, b] \]
		\end{theorembox}

\newpage
ในทำนองเดียวกันกับตัวหารร่วมมาก เราสามารถขยายบทนิยามของตัวคูณร่วมและตัวคูณร่วมน้อยของจำนวนเต็มมากกว่าสองจำนวนได้ดังนี้ 

\begin{definitionbox}
	ให้ $a_1, a_2, a_3, \ldots, a_n \in \mathbb{Z}$ ซึ่ง $a_i \neq 0$ สำหรับทุก $i = 1, 2, 3, \ldots, n$ เราเรียก $m \in \mathbb{N}$ ว่า \textbf{ตัวคูณร่วม} (common multiple) ของ $a_1, a_2, a_3, \ldots, a_n$ ก็ต่อเมื่อ $a_i \mid m$ สำหรับทุก $i = 1, 2, 3, \ldots, n$ 
	
	และเรียก $m$ ว่า \textbf{ตัวคูณร่วมน้อย} (least common multiple) ของ $a_1, a_2, a_3, \ldots, a_n$ ก็ต่อเมื่อ $m \in \mathbb{N}$ และ $m$ เป็นตัวคูณร่วมที่มีค่าน้อยที่สุด ซึ่งเขียนแทนด้วย $[a_1, a_2, a_3, \ldots, a_n]$
\end{definitionbox}

จากบทนิยาม จะพบว่า
\begin{enumerate}
	\item เราสามารถหา $[a_1, a_2, a_3, \ldots, a_n]$ ได้เสมอและมีเพียงค่าเดียวเท่านั้น
	\item $[a_1, a_2, a_3, \ldots, a_n] = [[a_1, a_2, \ldots, a_{n-1}], a_n] = [a_1, a_2, \ldots, [a_{n-1}, a_n]]$
	\item ถ้า $a_i \mid c$ สำหรับทุก $i = 1, 2, 3, \ldots, n$ แล้ว $[a_1, a_2, a_3, \ldots, a_n] \mid c$
\end{enumerate}

\begin{examplebox}
	\begin{enumerate}
		\item $[7, 11, 13] = 1001$
		\item สำหรับการหา $[3, 8, 10, 15]$ สามารถคำนวณตามลำดับได้ดังนี้
		\begin{align*}
			[3, 8, 10, 15] &= [[3, 8, 10], 15] \\
			&= [[[3, 8], 10], 15] \\
			&= [[24, 10], 15] \\
			&= [120, 15] \\
			&= 120
		\end{align*}
	\end{enumerate}
\end{examplebox}
